 #include <stdio.h>
#include <conio.h>
void main()
{
char com[100];
int i = 2, f = 0;
printf("\n Enter the line : ");
gets(com);
if (com[0] == '/')
{
if (com[1] == '/')
printf("\n It is a single line comment.\n\n");
else if (com[1] == '*')
{
for (i = 2; i <= 40; i++)
{
if (com[i] == '*' && com[i + 1] == '/')
{
printf("\n It is a multi-line comment.\n\n");
 f = 1;
break;
}
else
continue;
}
if (f == 0)
printf("\n It is not a comment.\n\n");
}
else
printf("\n It is not a comment.\n\n");
}
else
printf("\n It is not a comment.\n\n");
}
